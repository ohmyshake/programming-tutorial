"""

import os 
import subprocess
import glob
import shutil 
import pandas as pd
import numpy as np
from obspy.core import UTCDateTime


'''
è¯¥ç¨‹åºä¼šå°† file_path çš„ seed æ–‡ä»¶å…¨éƒ¨è§£å‹æˆsacï¼Œå¹¶å»é™¤ä»ªå™¨å“åº”æ·»åŠ åˆ° output_path ç›®å½•ä¸‹

## 1. è¾“å…¥å‚æ•° (è·¯å¾„å¿…é¡»æ˜¯å…¨è‹±æ–‡ï¼Œå¦åˆ™ subprocess.Popen ä¼šæŠ¥é”™)

file_path = './2_M_more_than_4/yunnan_network'                                 # è¾“å…¥æ–‡ä»¶å¤¹è·¯å¾„ yunnan_network  ||  instability_network
output_path = './sac_data'                                                      # è¾“å‡ºæ–‡ä»¶è·¯å¾„

readme_file = os.path.join(output_path,'readme')                                # è¾“å‡º logging æ–‡ä»¶è·¯å¾„

f1 = 0.001                                                                      # å»é™¤ä»ªå™¨å“åº”å¸Œæœ›çš„æ»¤æ³¢èŒƒå›´, å…ˆå»é™¤ä»ªå™¨å“åº”ï¼Œå†é™é‡‡æ ·
f2 = 0.003
f3 = 2.1
f4 = 2.3

delta = 0.2                                                                     # é™é‡‡æ ·åˆ° 0.2s




############################
## 2. è¯»å–åœ°éœ‡ç›®å½•ä¿¡æ¯
df = pd.read_excel('./fast_20210601232138.xls')
catalog_raw = df.values                                                         # pandas è½¬æˆ numpy

# catalog_new format: UTCtime/lat/lon/dep/mag
catalog_new = np.zeros(shape=catalog_raw.shape)
for i in range(0,catalog_raw.shape[0]):
    UTCtime = UTCDateTime(catalog_raw[i,0]+'T'+catalog_raw[i,1]+'+08')          # åœ°éœ‡ç›®å½•çš„æ—¶é—´éƒ½æ˜¯åŒ—äº¬æ—¶é—´ï¼Œæ—¶åŒºä¸º UTC+8ï¼Œéœ€è¦è½¬æ¢æˆ UTC æ—¶é—´ç»Ÿä¸€æ ‡å‡†
    nptime = UTCtime.timestamp
    lat = catalog_raw[i,2]
    lon = catalog_raw[i,3]
    dep = catalog_raw[i,4]
    mag = catalog_raw[i,5]

    catalog_new[i,0] = nptime
    catalog_new[i,1] = lat
    catalog_new[i,2] = lon

    ## åœ°éœ‡æ·±åº¦å¯èƒ½ä¸å­˜åœ¨ï¼Œè®¾æˆ999
    try:
        catalog_new[i,3] = float(dep)
    except:
        catalog_new[i,3] = 999

    ## åœ°éœ‡éœ‡çº§å¯èƒ½ä¸å­˜åœ¨ï¼Œè®¾æˆ999
    try:
        catalog_new[i,4] = float(mag)
    except:
        catalog_new[i,4] = 999



############################
## 3. rdseed to sac
seed_path = sorted(glob.glob( os.path.join(file_path,'*.seed') ))               # *.seed

if rm_old_output_path:                                                          # remove old output_path
    if os.path.exists(output_path):
        shutil.rmtree(output_path)
if os.path.exists(output_path) == False:                                          # mkdir new output_path
    os.makedirs(output_path)

fp = open(readme_file, 'w')
for i in range(0,len(seed_path)):
    seed_name = seed_path[i].split('/')[-1]                                     # seed_nameçš„æ—¶é—´æ˜¯åŒ—äº¬æ—¶é—´ï¼Œåªç²¾ç¡®åˆ°åˆ†é’Ÿ

    ### a.rdseed
    cmd_rdseed = subprocess.Popen(['rdseed', '-pdf',seed_name], cwd=file_path)   # rdseed è§£å‹æ–‡ä»¶, æ³¨æ„ seed_name åªèƒ½ç”¨ç›¸å¯¹è·¯å¾„ï¼Œå› ä¸ºpopenå‘½ä»¤åªä¼šæ‰¾å½“å‰è·¯å¾„ä¸‹çš„æ–‡ä»¶åç§°ï¼Œè¯¥æ–‡ä»¶åç§°ä¸åŒ…å«è·¯å¾„ä¿¡æ¯
    cmd_rdseed.wait()


    ### b.cat
    cmd_cat = subprocess.Popen('cat *_PZs_* >> SAC.PZs', shell=True, cwd=file_path) # åˆå¹¶å•ä¸ªseedæ–‡ä»¶æ‰€æœ‰çš„ä»ªå™¨å“åº”æ–‡ä»¶
    cmd_cat.wait()

    ### b.1 å°† rdseed ç”Ÿæˆçš„åŸå§‹ sac æ–‡ä»¶ä¿å­˜èµ·æ¥
    output_seed_path = os.path.join(output_path,'RAW_SAC',seed_name)
    if os.path.exists(output_seed_path)==False:                                 # mkdir new output seed path
        os.makedirs(output_seed_path)

    for file in glob.glob(file_path+'/*.SAC'):                                  # cp sac
        file_name = file.split('/')[-1]
        out_name = os.path.join(output_seed_path,file_name)
        shutil.copyfile(file, out_name)

    PZ_file = os.path.join(file_path,'SAC.PZs')                               # cp SAC.PZs
    shutil.copyfile(PZ_file, os.path.join(output_seed_path,'SAC.PZs'))


    ### c.sac
    os.putenv("SAC_DISPLAY_COPYRIGHT", '0')                                     # è®¾ç½®ç¯å¢ƒå˜é‡
    cmd_sac = subprocess.Popen(['sac'], stdin=subprocess.PIPE, cwd=file_path)   # åˆ©ç”¨sacç¨‹åºå»é™¤ä»ªå™¨å“åº” /Users/yf/1.Software/1.SAC/sac/bin/sac
    s = ''
    for sacfile in sorted(glob.glob(os.path.join(file_path,'*.SAC'))):
        sac_name = sacfile.split('/')[-1]                                       # sac_nameçš„æ—¶é—´æ˜¯UTCæ—¶é—´ï¼Œç²¾ç¡®åˆ°æ¯«ç§’

        year = int(sac_name.split('.')[0])                                      # é€šè¿‡sacæ–‡ä»¶åç§°è¯»å–æ—¶é—´ï¼Œåç»­å¯ä»¥ä¿®æ”¹æˆè¯»å– sac çš„å¤´éƒ¨å˜é‡
        julday = int(sac_name.split('.')[1])
        hour = int(sac_name.split('.')[2])
        minute = int(sac_name.split('.')[3])
        second = int(sac_name.split('.')[4])
        UTCtime = UTCDateTime(year=year, julday=julday, hour=hour, minute=minute, second=second) 
        nptime = UTCtime.timestamp

        evla = 0; evlo = 0; evdp = 0; mag = 0
        for j in range(catalog_new.shape[0]):
            if abs(catalog_new[j,0]-nptime) < 60:                               # ç²¾ç¡®åˆ°60ç§’, å› ä¸ºsacæ–‡ä»¶å¼€å§‹è®°å½•çš„æ—¶é—´ä¸æ˜¯åœ°éœ‡å¼€å§‹æ—¶åˆ»ï¼Œseedæ–‡ä»¶å¯¹ç§’è¿›è¡Œèˆå…¥åˆ°åˆ†é’Ÿ
                evla = catalog_new[j,1]
                evlo = catalog_new[j,2]
                evdp = catalog_new[j,3]
                mag = catalog_new[j,4]
                break

        s += "r %s \n" % sac_name 
        s += "rmean; rtr; taper \n" 
        s += "ch evla %f evlo %f evdp %f mag %f\n" % (evla,evlo,evdp,mag)
        s += "wh\n"
        s += "trans from polezero s SAC.PZs to vel freq %f %f %f %f\n" % (f1,f2,f3,f4)   # å»å®Œä»ªå™¨å“åº”å˜æˆé€Ÿåº¦è®°å½• to vel
        s += "mul 1.0e9 \n" 
        s += "lp c %f \n" % (1/(2*delta))
        s += "interpolate delta %f \n" % delta
        s += "w over \n"
    s += "q \n"

    stdout, stderr = cmd_sac.communicate(s.encode())                            # Popen.communicate å‘½ä»¤ä¼šè‡ªå¸¦ Popen.wait(), ä½†ä¿é™©èµ·è§è¿˜æ˜¯éœ€è¦è‡ªå·±æ·»åŠ  wait()å‘½ä»¤
    cmd_sac.wait()


    ### d.write logging readme
    if (evla==0) and (evlo==0) and (evdp==0) and (mag==0):                      # åˆ¤æ–­è¯¥seedäº‹ä»¶ï¼Œæ˜¯å¦åœ¨åœ°éœ‡ç›®å½•ä¸­æ‰¾åˆ°
        fp.write('not_find_in_catalog:   '+seed_name)
        fp.write('\n')
    else:
        fp.write('catalog_time:\t')                                             # åˆ¤æ–­è¯¥seedäº‹ä»¶,å­˜åœ¨åœ°éœ‡ç›®å½•
        fp.write(str(UTCDateTime(catalog_new[j,0]))+'\t')
        fp.write('sac_time:\t')
        fp.write(str(UTCDateTime(nptime))+'\t')
        fp.write('difference_time:\t')
        fp.write(str(catalog_new[j,0]-nptime)+'\t')
        fp.write('\n')


    ### e.remove file
    if (evla==0) and (evlo==0) and (evdp==0) and (mag==0):
        output_seed_path = os.path.join(output_path,'no_event_'+seed_name)      # æ²¡æœ‰åœ°éœ‡ç›®å½•çš„æ–‡ä»¶å¤¹åç§°
        if os.path.exists(output_seed_path)==False:                             # mkdir new output seed path
            os.makedirs(output_seed_path)
    else:
        output_seed_path = os.path.join(output_path,seed_name)
        if os.path.exists(output_seed_path)==False:                             # mkdir new output seed path
            os.makedirs(output_seed_path)

    for file in glob.glob(file_path+'/*.SAC'):                                  # cp
        file_name = file.split('/')[-1]
        out_name = os.path.join(output_seed_path,file_name)
        shutil.copyfile(file, out_name)

    for file in glob.glob(file_path+'/*.SAC'):                                  # remove sac file
        os.remove(file)
    for file in glob.glob(file_path+'/SAC_PZs_*'):                              # remove PZs file
        os.remove(file)
    os.remove(file_path+'/SAC.PZs')                                             # remove SAC.PZs


fp.close()
